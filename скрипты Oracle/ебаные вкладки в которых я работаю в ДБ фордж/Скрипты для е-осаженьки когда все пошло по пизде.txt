SELECT
  COUNT(e.POLICY_REC_ID),
  TRUNC(e.POLICY_CREATE_DATE,'hh24') 
  FROM RSA_EOSAGO.EPOLICY e 
  WHERE e.POLICY_CREATE_DATE >= TRUNC(SYSDATE)
  AND e.POLICY_STATE_ID = 1
  AND e.IS_EGARANT = 1
  GROUP BY
  TRUNC(e.POLICY_CREATE_DATE,'hh24')
  ORDER BY
  TRUNC(e.POLICY_CREATE_DATE,'hh24');
  
SELECT
  TRUNC(csel.CREATE_DATE,'hh24') "Разбивка по часам",
  COUNT(csel.DRAFT_POLICY_ID) "Кол-во мошенников"
  FROM RSA_EGARANT.CHECK_SCAN_EVENT_LOG csel 
  WHERE csel.CREATE_DATE >= TRUNC(SYSDATE)
  GROUP BY
  TRUNC(csel.CREATE_DATE,'hh24')
  ORDER BY
  TRUNC(csel.CREATE_DATE,'hh24');
